send_to_university_interaction = {
	category = interaction_category_friendly
	common_interaction = yes
	use_diplomatic_range = yes
	desc = send_to_university_interaction_desc

	auto_accept = yes

	is_shown = {
		scope:actor = {
			is_ruler = yes
		}
		scope:recipient = {
			is_adult = no
			OR = {
				guardian_university_access_trigger = yes
				scope:actor = {
					has_university_access_trigger = yes
				}
			}
		}
	}

	is_valid = {
		scope:actor = {
			gold >= monumental_gold_value
		}
		scope:recipient = {
			is_adult = no
			OR = {
				guardian_university_access_trigger = yes
				scope:actor = {
					has_university_access_trigger = yes
				}
			}
			already_sent_to_university_trigger = no
		}

	}

	on_accept = {
		scope:actor = {
			remove_short_term_gold = monumental_gold_value
		}
		scope:recipient = {
			custom_tooltip = is_sent_to_university
			add_character_flag = {
				flag = studying_at_university
				years = 17
			}
		}
	}
}
